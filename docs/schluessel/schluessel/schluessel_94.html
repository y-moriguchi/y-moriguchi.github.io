6.5 メタ循環Lispインタプリタ

メタ循環Lispインタプリタ、つまりLispを解析するLispのプログラムです。 

(define-syntax bind-vars
  (syntax-rules ()
    ((_ x (a b ...) (c ...) body ...)
      (if (pair? x)
          (let ((a (car x)))
            (bind-vars (cdr x) (b ...) (c ...) body ...))
          (error "Syntax Error")))
    ((_ x () (c d ...) body ...)
      (if (pair? x)
          (let ((c (car x)))
            (bind-vars (cdr x) () (d ...) body ...))
          (let ((c (if #f #f)))
            (bind-vars x () (d ...) body ...))))
    ((_ x () () body ...)
      (if (null? x)
          (begin body ...)
          (error "Syntax Error")))))

(define-syntax bind-vars-opt
  (syntax-rules ()
    ((_ x (a b ...) rest body ...)
      (if (pair? x)
          (let ((a (car x)))
            (bind-vars-opt (cdr x) (b ...) rest body ...))
          (error "Syntax Error")))
    ((_ x () rest body ...) (let ((rest x)) body ...))))

(define-record-type meta-closure
  (make-meta-closure vars body)
  meta-closure?
  (vars get-vars)
  (body get-body))

(define *global-environment*
  `((null? . ,null?) (eq?   . ,eq?)  (car   . ,car)  (cdr   . ,cdr)
    (cons  . ,cons)  (+     . ,+)    (-     . ,-)    (*     . ,*)
    (/     . ,/)     (=     . ,=)))

(define (bind-to-env name val env) (cons name val))

(define (bind-all adef args env)
  (cond ((and (null? adef) (null? args)) env)
        ((or (null? adef) (null? args))
          (error "Illegal Argument"))
        ((symbol? adef) (cons (cons adef args) env))
        ((and (pair? adef) (pair? args))
          (bind-all (cdr adef) (cdr args)
                    (cons (cons (car adef) (car args)) env)))
        (else (error "Illegal Argument"))))

(define (find-from-env name env)
  (cdr (or (assq name env)
           (assq name *global-environment*)
           (error "Unbound Symbol"))))

(define (eval-meta x env)
  (cond ((symbol? x) (find-from-env x env))
        ((not (pair? x)) x)
        ((eq? (car x) 'if) (eval-if (cdr x) env))
        ((eq? (car x) 'lambda) (eval-lambda (cdr x) env))
        ((eq? (car x) 'define) (eval-define (cdr x) env))
        ((eq? (car x) 'begin) (eval-begin (cdr x) env))
        ((eq? (car x) 'quote) (eval-quote (cdr x) env))
        (else
          (apply-meta
            (eval-meta (car x) env)
            (map (lambda (x) (eval-meta x env)) (cdr x))
            env))))

(define (eval-if x env)
  (bind-vars x (cnd tr) (fl)
    (if (eval-meta cnd env)
        (eval-meta tr env)
        (eval-meta fl env))))

(define (eval-lambda x env)
  (bind-vars-opt x (vars) body
    (make-meta-closure vars body)))

(define (eval-define x env)
  (cond ((and (pair? x) (pair? (car x)))
          (bind-vars-opt x (vars) body
            (bind-vars-opt vars (name) args
              (set! *global-environment*
                    (cons
                      (bind-to-env
                        name
                        (make-meta-closure args body)
                        env)
                      *global-environment*))
              name)))
        (else
          (bind-vars x (name val) ()
            (set! *global-environment*
                  (cons
                    (bind-to-env name (eval-meta val env) env)
                    *global-environment*))
            name))))

(define (eval-begin f env)
  (cond ((null? f) r)
        ((null? (cdr f)) (eval-meta (car f) env))
        (else
          (eval-meta (car f) env)
          (eval-begin (cdr f) env))))

(define (eval-quote args env)
  (if (and (pair? args))
      (car args)
      (error "invalid quote")))

(define (apply-meta cl args env)
  (cond ((procedure? cl) (apply cl args))
        ((meta-closure? cl)
          (eval-begin
          (get-body cl)
          (bind-all (get-vars cl) args env)))
        (else  (error "Cannot apply"))))
