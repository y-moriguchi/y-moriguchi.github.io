6.1 モンテカルロ積分

乱数により積分の値を見積もる方法です。 (SICPの問題3.5の解答になっています。) 

(define (random-in-range low high)
  (+ (* (random-real) (- high low)) low))

(define (random-vals vals)
  (let loop ((d vals))
    (if (null? d)
        '()
        (let ((x (car d)))
          (cons
            (cons (car x) (random-in-range (cadr x) (caddr x)))
            (loop (cdr d)))))))

(define (eval-expr expr vals)
  (let ((a (random-vals vals)))
    (eval expr
         (make-environment a (scheme-report-environment 5)))))

(define (get-result vals k c)
  (let loop ((d vals) (area 1))
    (if (null? d)
        (exact->inexact (* area (/ c k)))
        (let ((x (car d)))
          (loop (cdr d) (* area (- (caddr x) (cadr x))))))))

(define (estimate-integral expr vals n)
  (let loop ((k n) (c 0))
    (cond ((zero? k) (get-result vals n c))
          ((eval-expr expr vals) (loop (- k 1) (+ c 1)))
          (else (loop (- k 1) c)))))

(estimate-integral
  '(<= (+ (* x x) (* y y)) 1)
  '((x -1 1) (y -1 1))
  5000)
