6.4 RLC回路のシミュレート

ストリームを使用してRLC回路のシミュレートをするプログラムです。 (SICPの問題3.80の解答になっています。) 

(use srfi-41)
(define (add-streams s1 s2) (stream-map + s1 s2))
(define (scale-stream stream factor)
  (stream-map (lambda (x) (* x factor)) stream))
(define (display-stream stream n)
  (if (zero? n)
      'ok
      (begin
        (print (stream-car stream))
        (display-stream (stream-cdr stream) (- n 1)))))

(define (integral delayed-integrand initial-value dx)
  (define int
    (stream-cons initial-value
                 (let ((integrand (force delayed-integrand)))
                   (add-streams (scale-stream integrand dx) int))))
  int)

(define (RLC r l c dt)
  (lambda (il0 vc0)
    (define vc (integral (delay dvc) vc0 dt))
    (define il (integral (delay dil) il0 dt))
    (define dvc (scale-stream il (/ -1 c)))
    (define dil (add-streams (scale-stream vc (/ l))
                              (scale-stream il (- (/ r l)))))
    (stream-map (lambda (x y) (cons x y)) vc il)))
(define z (RLC 1 1 0.2 0.1))
(define x (z 0 10))
(display-stream (stream-map car x) 100)
