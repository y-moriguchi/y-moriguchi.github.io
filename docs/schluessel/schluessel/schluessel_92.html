6.3 ハフマン符号

ハフマン符号でオブジェクトの列を符号化するサンプルです。 (SICPの問題2.68〜2.70の解答になっています。) 

(define (make-leaf symbol weight)
  (list 'leaf symbol weight))
(define (leaf? o) (and (pair? o) (eq? (car o) 'leaf)))
(define (symbol-leaf x) (cadr x))
(define (weight-leaf x) (caddr x))

(define (make-code-tree left right)
  (list left
       right
       (append (symbols left) (symbols right))
       (+ (weight left) (weight right))))
(define (left-branch x) (car x))
(define (right-branch x) (cadr x))
(define (symbols tree)
  (if (leaf? tree)
      (list (symbol-leaf tree))
      (caddr tree)))
(define (weight tree)
  (if (leaf? tree)
      (weight-leaf tree)
      (cadddr tree)))

(define (decode bits tree)
  (let loop ((bits bits) (branch tree))
    (if (null? bits)
        '()
        (let ((next-branch (choose-branch (car bits) branch)))
          (if (leaf? next-branch)
              (cons (symbol-leaf next-branch)
                   (loop (cdr bits) tree))
              (loop (cdr bits) next-branch))))))
(define (choose-branch bit branch)
  (cons ((= bit 0) (left-branch branch))
       ((= bit 1) (right-branch branch))
       (else (error "bad bit"))))

(define (adjoin-set x set)
  (cond ((null? set) (list x))
        ((< (weight x) (weight (car set))) (cons x set))
        (else (cons (car set)
                    (adjoin-set x (cdr set))))))
(define (make-leaf-set pairs)
  (if (null? pairs)
      '()
      (let ((pair (car pairs)))
        (adjoin-set (make-leaf (car pair) (cadr pair))
                    (make-leaf-set (cdr pairs))))))

(define (encode message tree)
  (if (null? message)
      '()
      (append (encode-symbol (car message) tree)
              (encode (cdr message) tree))))
(define (encode-symbol symbol tree)
  (if (not (memv symbol (symbols tree)))
      (error "cannot encode the symbol"))
  (let loop ((branch tree))
    (if (leaf? branch)
        '()
        (let ((lbranch (left-branch branch))
              (rbranch (right-branch branch)))
          (if (memv symbol (symbols lbranch))
              (cons 0 (loop lbranch))
              (cons 1 (loop rbranch)))))))

(define (generate-huffman-tree pairs)
  (succesive-merge (make-leaf-set pairs)))
(define (succesive-merge set)
  (cond ((null? set) (error "cannot null"))
        ((null? (cdr set)) (car set))
        (else (succesive-merge
                (adjoin-set (make-code-tree (car set) (cadr set))
                            (cddr set))))))

(define
  words
  '(get a job
    sha na na na na na na na na
    get a job
    sha na na na na na na na na
    wah yip yip yip yip yip yip yip yip yip
    sha boom))
(define
  words-pairs
  '((a 2) (na 16) (boom 1) (sha 3) (get 2) (yip 9) (job 2) (wah 1)))

(define t0 (generate-huffman-tree words-pairs))
(encode words t0)
