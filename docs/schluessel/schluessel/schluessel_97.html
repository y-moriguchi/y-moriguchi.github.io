6.8 メモ帳

SchluesselのSwing連携機能を用いたメモ帳です。 

(define (save-as)
  (let ((rs (show-save-file-chooser *chooser* *window1*)))
    (and rs
         (begin
           (set! *filepath* rs)
           (save-component *memopad* *filepath*)))))

(define (open-as)
  (let ((rs (show-open-file-chooser *chooser* *window1*)))
    (and rs
         (begin
           (set! *filepath* rs)
           (load-component *memopad* *filepath*)))))

(define *window1*
  (make-window
    "Memopad" 400 400
    `(("File" (("New"  ,(lambda (e)
                          (set-text! *memopad* "")
                          (set! *filepath* #f)))
               ("Open" ,(lambda (e) (open-as)))
               ("Save"
                ,(lambda (e)
                   (if *filepath*
                       (save-component *memopad* *filepath*)
                       (save-as))))
               ("Save as" ,(lambda (e) (save-as)))
               separator
               ;("Print" ,(lambda (e) (print-component *memopad*)))
               ("Exit" ,(lambda (e) (exit 0)))))
      ("Help"
       ,(lambda (e)
          (show-message-dialog #f "memopad" "Help" 'plain))))))

(define *filepath* #f)
(define *memopad*  (make-text-pane))
(define *chooser*
  (make-file-chooser '(("txt" . "Text file"))))

(layout-border
  *window1*
  `((center . ,(make-scroll-pane *memopad*))))
(show-window *window1*)
